# Formato: INPUT → OUTPUT_ESPERADO
# Líneas que empiezan con # son comentarios

# ========================================
# PRUEBAS BÁSICAS - Desplazamientos pequeños
# ========================================

# Desplazamiento 5

# PRUEBAS DE CICLO (MOD 26) - Wraparound
